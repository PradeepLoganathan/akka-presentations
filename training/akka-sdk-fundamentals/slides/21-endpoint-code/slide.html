api/CustomerEndpoint.java
@HttpEndpoint("/api")
@Acl(allow = @Acl.Matcher(principal = Acl.Principal.INTERNET))
public class CustomerEndpoint {

  private final ComponentClient client;

  public CustomerEndpoint(ComponentClient client) {
    this.client = client;
  }

  @Post("/customers")
  public CreateCustomerResponse create(CreateCustomerRequest req) {
    var cmd = new CreateCustomer(req.firstName, req.lastName);
    client.forEventSourcedEntity(req.customerId)
        .method(CustomerEntity::create)
        .invoke(cmd);
    return new CreateCustomerResponse(
        req.customerId, req.firstName, req.lastName);
  }

  @Get("/customers")
  public customerEntry list() {
    return client.forView().method(CustomerView::getAll).invoke();
  }
}
  • ComponentClient is how endpoints call components
  • Injected by the runtime — no DI setup
  • forEventSourcedEntity(id) targets one entity instance
  • forView() queries a view
  • .method(…::…) is a method reference to the handler
  • Request & response records are plain Java
