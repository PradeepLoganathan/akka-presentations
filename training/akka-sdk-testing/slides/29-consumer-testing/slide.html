var tk = ConsumerTestKit.of(CustomerAuditConsumer::new);

var result = tk.onEvent(
    new customerCreated("c-1", "Ada", "Lovelace"));

// side effects captured for assertion
var calls = result.getSideEffects();
assertThat(calls).hasSize(1);
assertThat(calls.get(0).targetComponent()).isEqualTo("audit-log");
