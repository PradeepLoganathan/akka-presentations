@ComponentId("order-workflow")
public class OrderWorkflow
    extends Workflow<OrderWorkflow.State> {

  public record State(
      String orderId, String customerId,
      List<OrderLine> lines, long totalCents) {}

  private final ComponentClient client;

  public OrderWorkflow(ComponentClient client) {
    this.client = client;
  }

  @Override
  public WorkflowDef<State> definition() {
    return workflow()
        .addStep(reserveStep())
        .addStep(chargeStep())
        .addStep(shipStep())
        .addStep(compensateStep());
  }
}
  • @ComponentId like any component
  • Workflow<State> - typed on the workflow's own state
  • Constructor accepts ComponentClient to call entities
  • definition() lists the steps in scope
  • Method references keep the code shape narrow
